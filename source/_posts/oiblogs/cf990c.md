就是分成四类：
- 好序列
- 需要**若干**左括号的序列
- 需要**若干**右括号的序列
- 凉透了的废材序列

拼一拼就行了

```cpp
#include <iostream>
#include <string>
#include <map>
using namespace std;
int n, cnt[5];
map<int,int> mp;
string a;
void isLegal(string a){
	int din=0, len=a.length(), notSaveAble=0;
	for(int i=0; i<len; i++){
		if(a[i]=='('){
			if(din<0){
				notSaveAble += -din;
				din = 0;
			}
			din++;
		}
		else
			din--;
	}
	if(din<0){
		notSaveAble += -din;
		din = 0;
	}
	if(!notSaveAble && !din)	mp[0]++;
	else if(!notSaveAble && din)	mp[din]++;
	else if(notSaveAble && !din)	mp[-notSaveAble]++;
}
int main(){
	cin>>n;
	while(n--){
		cin>>a;
		isLegal(a);
	}
	long long ans=0;
	for(int i=0; i<=300000; i++)
		ans += (long long)mp[i] * mp[-i];
	cout<<ans<<endl;
	return 0;
}
```
