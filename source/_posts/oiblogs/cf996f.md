大力猜结论：所有位于定义域内的数的取值的概率都相等，所以答案就是平均数。

<!--more-->

[ref](https://blog.csdn.net/ZscDst/article/details/80832919)

```cpp
#include <iostream>
#include <cstdio>
using namespace std;
typedef long long ll;
int n, r, a[262155], uu, vv;
ll sum;
int main(){
	cin>>n>>r;
	for(int i=0; i<(1<<n); i++){
		scanf("%d", &a[i]);
		sum += a[i];
	}
	printf("%.12f\n", (double)sum/(1<<n));
	while(r--){
		scanf("%d %d", &uu, &vv);
		sum -= a[uu];
		a[uu] = vv;
		sum += a[uu];
		printf("%.12f\n", (double)sum/(1<<n));
	}
	return 0;
}
```
