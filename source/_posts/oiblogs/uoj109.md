#include <algorithm>
#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <vector>
#include <ctime>
#include <map>
using namespace std;
typedef pair<int,int> par;
map<par,int> mp;
vector<par> vec;
int rnd(int f){
	return (long long)rand()*rand()%f;
}
int main(){
	srand(time(NULL));
	for(int i=1; i<999; i++){
		int f=rnd(i);
		vec.push_back(par(f, i));
		mp[par(f,i)] = 1;
	}
	int tmp=1501-998;
	while(tmp){
		int a=rnd(999), b=rnd(999);
		if(a>b)	swap(a, b);
		if(mp[par(a,b)])	continue;
		mp[par(a,b)] = 1;
		vec.push_back(par(a,b));
		tmp--;
	}
	cout<<"999 "<<vec.size()<<endl;
	for(int i=0; i<vec.size(); i++)
		cout<<vec[i].first<<" "<<vec[i].second<<endl;
	return 0;
}

#include <iostream>
#include <cstdio>
using namespace std;
int main(){
	cout<<"999 1501\n";
	//0:0
	//1-499:1
	//500-998:0
	for(int i=1; i<=499; i++)
		cout<<"0 "<<i<<endl;
	for(int i=1; i<=499; i++)
		cout<<i<<" "<<i+499<<endl;
	int rev=1501-499*2;
	for(int i=1; i<=499 && rev; i++){
		for(int j=500; j<=998 && rev; j++)
			if(i!=j-499){
				cout<<i<<" "<<j<<endl;
				rev--;
			}
	}
	return 0;
}
