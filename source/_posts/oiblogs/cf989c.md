#include <iostream>
#include <cstdio>
using namespace std;
char ss[55][55];
int a, b, c, d;
void init(int u, char v){
	for(int i=u; i<=u+12; i++)
		for(int j=1; j<=50; j++)
			ss[i][j] = v;
}
void build(int u, int v, char w){
	for(int i=u+1; ; i+=2){
		int j;
		j = 2;
		for(; v && j<50; j+=2){
			ss[i][j] = w;
			v--;
		}
		if(!v)	break;
	}
}
int main(){
	cin>>a>>b>>c>>d;
	a--; b--; c--; d--;
	cout<<"48 50\n";
	for(int i=0; i<4; i++)
		init(i*12+1, 'A'+i);
	build(1, b, 'B');
	build(13, c, 'C');
	build(25, d, 'D');
	build(37, a, 'A');
	for(int i=1; i<=48; i++){
		for(int j=1; j<=50; j++)
			putchar(ss[i][j]);
		printf("\n");
	}
	return 0;
}
